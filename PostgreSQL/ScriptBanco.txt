CREATE SCHEMA dbmcs001 AUTHORIZATION postgres;

  CREATE TABLE dbmcs001.tbusuario
(
  codigo integer NOT NULL,
  nome character varying(100) NOT NULL DEFAULT ''::character varying,
  login character varying(100) NOT NULL DEFAULT ''::character varying,
  senha character varying(100) NOT NULL DEFAULT ''::character varying,
  situacao integer NOT NULL,
  CONSTRAINT tbusuario_pkey PRIMARY KEY (codigo)
)
WITH (
  OIDS=FALSE
);

CREATE TABLE dbmcs001.tbpermissao
(
  codigo integer NOT NULL,
  codusuario integer NOT NULL,
  permissao character varying(100) NOT NULL DEFAULT ''::character varying,
  CONSTRAINT tbpermissao_pkey PRIMARY KEY (codigo, permissao),
  CONSTRAINT tbpermissao_fkey FOREIGN KEY (codusuario)
      REFERENCES dbmcs001.tbusuario (codigo) MATCH SIMPLE
      ON UPDATE NO ACTION ON DELETE NO ACTION
)
WITH (
  OIDS=FALSE
);
